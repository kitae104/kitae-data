CSV에서 숫자가 "51,248,233"처럼 쉼표가 포함된 문자열로 읽혀집니다. 계산을 하려면 쉼표를 제거하고 정수로 변환해야 합니다.

cell_03_clean.py 
# ① 숫자 컬럼 전처리: 쉼표 제거 후 숫자 타입으로 변환
df_clean = df.copy()   # 원본 보존

for col in df_clean.columns[1:]:   # 행정구역 열 제외
    df_clean[col] = (df_clean[col]
                     .astype(str)
                     .str.replace(',', '')   # 쉼표 제거
                     .str.strip())             # 앞뒤 공백 제거
    df_clean[col] = pd.to_numeric(df_clean[col], errors='coerce')

# ② 행정구역 이름 깔끔하게 정리 (코드 번호 제거)
# "서울특별시  (1100000000)" → "서울특별시"
df_clean['지역명'] = df_clean['행정구역'].str.split('  ').str[0]

print("정제 완료!")
print("지역 목록:", df_clean['지역명'].tolist())
출력 결과정제 완료! 지역 목록: ['전국', '서울특별시', '부산광역시', '대구광역시', '인천광역시', '광주광역시', '대전광역시', '울산광역시', '세종특별자치시', '경기도', '강원특별자치도', '충청북도', '충청남도', '전북특별자치도', '전라남도', '경상북도', '경상남도', '제주특별자치도']

🔑 핵심 패턴: str → 숫자 변환

str.replace(',', '') → 쉼표 제거
pd.to_numeric(…, errors='coerce') → 숫자로 변환 (변환 불가 시 NaN)
str.split(' ').str[0] → 공백 두 개로 쪼개서 첫 번째 부분만 가져오기
