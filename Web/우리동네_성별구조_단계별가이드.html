파일 읽기 — csv 모듈로 시작하기

원본 코드처럼 csv 모듈을 써서 data1.csv를 읽어봅니다

원본 노트북은 Python의 기본 내장 csv 모듈을 사용합니다. pandas를 쓰지 않아도 CSV를 읽을 수 있지만, 데이터를 하나하나 직접 처리해야 합니다. 먼저 이 방법으로 시작해 봅니다.
step1_read_csv.py — 파일 열고 내용 확인 
# ① csv 모듈 불러오기
import csv

# ② 파일 열기 (인코딩 지정 필수!)
#    cp949 = 한글 윈도우 기본 인코딩
f = open("./data1.csv", encoding='cp949')
data = csv.reader(f)

# ③ 첫 번째 행(헤더)만 출력해서 구조 확인
header = next(data)   # 첫 행 한 줄 읽기
print(f"총 컬럼 수: {len(header)}")
print("앞 5개 컬럼:", header[:5])

f.close()  # 파일 닫기
총 컬럼 수: 310 앞 5개 컬럼: ['행정구역', '2024년09월_계_총인구수', '2024년09월_계_연령구간인구수', '2024년09월_계_0세', '2024년09월_계_1세']
💡

원본 코드에는 encoding 지정이 없어서 오류가 날 수 있습니다. open("./data1.csv", encoding='cp949')처럼 인코딩을 명시하면 한글이 정상 출력됩니다.

⚠️

원본 코드: f = open("./data1.csv")
수정 코드: f = open("./data1.csv", encoding='cp949')
파일을 열면 반드시 f.close()로 닫거나, with open(...) as f: 구문을 씁니다.

step2_search.py — 지역명 검색 
import csv

# ① 찾을 지역명 입력받기
findDong = input("동/구/시 이름을 입력하세요: ")

# ② 파일을 열고 검색
with open("./data1.csv", encoding='cp949') as f:
    data = csv.reader(f)

    found = False   # 찾았는지 여부
    for row in data:
        if findDong in row[0]:   # 행정구역 이름에 포함되면
            print(f"찾았습니다! → {row[0]}")
            print(f"총인구: {row[1]}")
            found = True
            break   # 하나 찾으면 멈추기

    if not found:
        print(f"'{findDong}' 을(를) 포함한 지역을 찾지 못했습니다.")
동/구/시 이름을 입력하세요: 서울 찾았습니다! → 서울특별시 (1100000000) 총인구: 9,350,995

데이터 추출 — 남성·여성 인구 리스트 만들기

해당 행에서 연령별 남녀 인구를 숫자 리스트로 추출합니다

컬럼 위치 이해하기
0 = 행정구역 | 1 = 계_총인구수 | 2 = 계_연령구간인구수 | 3 = 계_0세 | … | 104 = 계_101 | 105 = 남_총인구수 | 106 = 남_0세 | … | 207 = 남_100세이상 | 208 = 여_총인구수 | 209 = 여_연령구간인구수 | 210 = 여_0세 | …
step3_extract.py — 남녀 인구 추출 
import csv

m = []   # 남성 인구 리스트 (음수로 저장 — 왼쪽 방향 막대를 위해)
f2 = []  # 여성 인구 리스트

findDong = input("동 이름 : ")

with open("./data1.csv", encoding='cp949') as f:
    data = csv.reader(f)
    for row in data:
        if findDong in row[0]:

            # 남성: 열 106 ~ 206 (0세~100세) → 음수로 저장
            for i in row[106:207]:
                i_clean = i.replace(',', '').strip()  # 쉼표 제거
                m.append(-int(i_clean))              # 음수로 변환

            # 여성: 열 210 ~ 310 (0세~100세) → 양수로 저장
            for i in row[210:]:
                i_clean = i.replace(',', '').strip()
                f2.append(int(i_clean))

            break

print(f"남성 데이터 개수: {len(m)}")
print(f"여성 데이터 개수: {len(f2)}")
print(f"남성 0세: {abs(m[0]):,}명 / 여성 0세: {f2[0]:,}명")
동 이름 : 서울 남성 데이터 개수: 101 여성 데이터 개수: 101 남성 0세: 19,407명 / 여성 0세: 18,693명
💡

왜 남성을 음수로 저장할까요?
막대 그래프에서 왼쪽(-) 방향으로 남성, 오른쪽(+) 방향으로 여성을 그려 인구 피라미드(항아리 그래프) 모양을 만들기 위해서입니다.

❌ 원본 코드 (쉼표 처리 없음)
for i in row[3:104]:
    m.append(-int(i))
# → 쉼표 포함 문자열이면
#   ValueError 오류 발생!
✅ 개선 코드 (쉼표 제거 추가)
for i in row[106:207]:
    i_clean = i.replace(',','').strip()
    m.append(-int(i_clean))
# → 쉼표 제거 후 변환
#   정상 작동!

그래프 그리기 — matplotlib으로 인구 피라미드 완성

barh()로 가로 막대 그래프를 그려 인구 구조를 시각화합니다

barh() = 가로 막대 그래프(horizontal bar chart)
range(101)은 0~100세, 남성은 음수 값이라 왼쪽으로, 여성은 양수라 오른쪽으로 그려집니다. 두 막대를 겹쳐서 표시합니다.
step4_chart_basic.py — 기본 인구 피라미드 
import matplotlib.pyplot as plt
import matplotlib

# 한글 폰트 (Windows)
matplotlib.rc('font', family='Malgun Gothic')
matplotlib.rcParams['axes.unicode_minus'] = False

# m, f2, findDong 은 앞 단계에서 이미 만들어져 있어야 함

plt.figure(figsize=(10, 8))
plt.title(findDong + " 지역의 남녀 성별 인구 분포",
        fontsize=14, fontweight='bold')

# 남성 막대 (음수 → 왼쪽으로)
plt.barh(range(101), m, label='남성', color='#3B82F6', alpha=0.8)

# 여성 막대 (양수 → 오른쪽으로)
plt.barh(range(101), f2, label='여성', color='#EF4444', alpha=0.8)

plt.xlabel('인구수 (명) — 왼쪽: 남성, 오른쪽: 여성')
plt.ylabel('나이 (세)')
plt.legend(fontsize=12)
plt.tight_layout()
plt.show()

pandas 버전 — 더 간결하고 안전
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib

matplotlib.rc('font', family='Malgun Gothic')
matplotlib.rcParams['axes.unicode_minus'] = False

# pandas로 읽기 (쉼표 자동 처리 설정)
df = pd.read_csv('data1.csv', encoding='cp949',
                  thousands=',')   # thousands=',' 로 자동 쉼표 제거!

findDong = input("동 이름 : ")

# 해당 행 찾기
row = df[df['행정구역'].str.contains(findDong)]
if len(row) == 0:
    print("지역을 찾지 못했습니다.")
else:
    r = row.iloc[0]

    # 남성 데이터 (음수)
    m_cols = [f'2024년09월_남_{i}세' for i in range(101)]
    m = [-r[c] for c in m_cols if c in r.index]

    # 여성 데이터
    f_cols = [f'2024년09월_여_{i}세' for i in range(101)]
    f2 = [r[c] for c in f_cols if c in r.index]

    fig, ax = plt.subplots(figsize=(10, 9))
    ax.barh(range(len(m)), m,  label='남성', color='#3B82F6', alpha=0.8)
    ax.barh(range(len(f2)), f2, label='여성', color='#EF4444', alpha=0.8)
    ax.axvline(0, color='black', linewidth=1)
    ax.set_title(findDong + " 지역의 남녀 성별 인구 분포",
                 fontsize=14, fontweight='bold')
    ax.set_xlabel('인구수 (명)')
    ax.set_ylabel('나이 (세)')
    ax.legend()
    plt.tight_layout()
    plt.show()

🎯 최종 완성 코드 (pandas 버전)

우리동네_성별구조_완성.py 
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# 우리동네 성별 인구 구조 시각화
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib
import platform

# ① 한글 폰트 설정 (OS별)
if platform.system() == 'Windows':
    matplotlib.rc('font', family='Malgun Gothic')
elif platform.system() == 'Darwin':  # Mac
    matplotlib.rc('font', family='AppleGothic')
matplotlib.rcParams['axes.unicode_minus'] = False

# ② 데이터 불러오기
df = pd.read_csv('data1.csv', encoding='cp949', thousands=',')

# ③ 지역명 입력 & 검색
findDong = input("동/구/시 이름을 입력하세요: ")
matched = df[df['행정구역'].str.contains(findDong, na=False)]

if len(matched) == 0:
    print(f"'{findDong}' 지역을 찾지 못했습니다.")
else:
    r = matched.iloc[0]
    region_name = r['행정구역'].split('  ')[0]

    # ④ 남녀 연령별 데이터 추출 (0~100세)
    ages = list(range(101))
    m_data  = []  # 남성 (음수)
    f_data  = []  # 여성 (양수)

    for i in ages:
        col_m = f'2024년09월_남_{i}세'
        col_f = f'2024년09월_여_{i}세'
        # 100세는 컬럼명이 '100세 이상' 이므로 예외 처리
        if i == 100:
            col_m = '2024년09월_남_100세 이상'
            col_f = '2024년09월_여_100세 이상'
        m_data.append(-r[col_m] if col_m in r.index else 0)
        f_data.append(r[col_f]  if col_f in r.index else 0)

    # ⑤ 그래프 그리기
    fig, ax = plt.subplots(figsize=(10, 10))

    ax.barh(ages, m_data, label='남성', color='#3B82F6', alpha=0.8)
    ax.barh(ages, f_data, label='여성', color='#EF4444', alpha=0.8)
    ax.axvline(0, color='gray', linewidth=1.2)   # 중심선

    # X축 절댓값 표시
    ticks = ax.get_xticks()
    ax.set_xticklabels([f'{int(abs(t)):,}' for t in ticks])

    ax.set_title(region_name + " 지역의 남녀 성별 인구 분포",
                fontsize=15, fontweight='bold', pad=15)
    ax.set_xlabel('인구수 (명)  ← 남성 | 여성 →')
    ax.set_ylabel('나이 (세)')
    ax.legend(fontsize=12, loc='lower right')
    ax.grid(axis='x', alpha=0.3)
    plt.tight_layout()
    plt.show()
