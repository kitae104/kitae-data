데이터 추출 — 연령별 인구 리스트 만들기

선택한 행에서 0세~100세 이상 인구를 리스트로 추출

컬럼명 패턴: 2026년01월_계_{나이}세
0세 → '2026년01월_계_0세'
30세 → '2026년01월_계_30세'
100세 이상 → '2026년01월_계_100세 이상' (띄어쓰기 주의!)
step3_extract.py — 연령별 인구 추출 
# 0세~100세 이상 인구를 리스트로 만들기
ages = list(range(101))   # [0, 1, 2, ..., 100]
pop  = []               # 각 나이별 인구를 담을 빈 리스트

for i in ages:
    if i < 100:
        col = f'2026년01월_계_{i}세'      # 예) '2026년01월_계_30세'
    else:
        col = '2026년01월_계_100세 이상'   # 100세 이상은 이름이 다름!
    pop.append(r[col])

print(f"총 연령 항목 수: {len(pop)}")
print(f"0세  인구: {pop[0]:,}명")
print(f"30세 인구: {pop[30]:,}명")
print(f"100세 이상: {pop[100]:,}명")
총 연령 항목 수: 101 0세 인구: 119명 30세 인구: 553명 100세 이상: 2명
💡

data1.csv에는 전체(계) 인구 컬럼만 들어있으므로
연령별 인구 리스트도 하나만 만들면 됩니다.
값을 그대로 정수로 저장해 두면 다음 단계에서 바로 그래프로 그릴 수 있습니다.

step4_chart_grouped.py — 10세 단위 묶음 버전
# 10세 단위로 묶어서 집계
group_labels = []
group_pop    = []

for start in range(0, 100, 10):
    end   = start + 9
    label = f'{start}~{end}세'
    total = sum(pop[start:end+1])   # 해당 구간 합계
    group_labels.append(label)
    group_pop.append(total)

# 100세 이상 추가
group_labels.append('100세\n이상')
group_pop.append(pop[100])

fig, ax = plt.subplots(figsize=(10, 7))

bars = ax.barh(group_labels, group_pop,
               color='#059669', alpha=0.85, edgecolor='white')

# 막대 끝에 숫자 표시
for bar, val in zip(bars, group_pop):
    ax.text(val + 5, bar.get_y() + bar.get_height()/2,
            f'{val:,}', va='center', fontsize=9)

ax.set_title(region_name + " 연령대별 인구 분포 (10세 단위)",
            fontsize=14, fontweight='bold')
ax.set_xlabel('인구수 (명)')
ax.grid(axis='x', alpha=0.3)
plt.tight_layout()
plt.show()

🎯 최종 완성 코드 (전체 통합)

우리동네_연령구조_완성.py 
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# 우리동네 연령별 인구 구조 시각화 (data1.csv)
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib
import platform

# ① 한글 폰트 설정
if platform.system() == 'Windows':
    matplotlib.rc('font', family='Malgun Gothic')
elif platform.system() == 'Darwin':
    matplotlib.rc('font', family='AppleGothic')
else:   # Google Colab / Linux
    import subprocess
    subprocess.run(['apt-get', '-qq', 'install', 'fonts-nanum'], capture_output=True)
    matplotlib.rc('font', family='NanumGothic')
matplotlib.rcParams['axes.unicode_minus'] = False

# ② 데이터 불러오기 (숫자에 쉼표 없음 → thousands 옵션 불필요)
df = pd.read_csv('data1.csv', encoding='cp949')

# ③ 지역명 입력 & 검색
findDong = input("지역 이름을 입력하세요 (예: 역삼1동, 강남구, 서울): ")
matched  = df[df['행정구역'].str.contains(findDong, na=False)]

if len(matched) == 0:
    print(f"'{findDong}' 지역을 찾지 못했습니다.")

else:
    # ④ 결과가 여러 개이면 목록 출력 후 선택
    if len(matched) > 1:
        print(f"'{findDong}' 검색 결과 ({len(matched)}건):")
        for i, (_, row) in enumerate(matched.iterrows()):
            name = row['행정구역']
            pop_total = row['2026년01월_계_총인구수']
            print(f"  [{i}] {name}  ({pop_total:,}명)")
        idx = int(input("번호를 선택하세요: "))
        r = matched.iloc[idx]
    else:
        r = matched.iloc[0]

    # ⑤ 지역명 정리 (코드 괄호 제거)
    region_name = r['행정구역'].split('(')[0].strip()

    # ⑥ 연령별 인구 리스트 만들기 (0세~100세이상)
    ages = list(range(101))
    pop  = []
    for i in ages:
        col = f'2026년01월_계_{i}세' if i < 100 else '2026년01월_계_100세 이상'
        pop.append(r[col])

    # ⑦ 10세 단위로 묶기
    group_labels, group_pop = [], []
    for start in range(0, 100, 10):
        group_labels.append(f'{start}~{start+9}세')
        group_pop.append(sum(pop[start:start+10]))
    group_labels.append('100세 이상')
    group_pop.append(pop[100])

    # ⑧ 그래프 그리기
    fig, ax = plt.subplots(figsize=(10, 7))
    bars = ax.barh(group_labels, group_pop,
                   color='#059669', alpha=0.85, edgecolor='white')

    for bar, val in zip(bars, group_pop):
        ax.text(val + val*0.01, bar.get_y() + bar.get_height()/2,
                f'{val:,}', va='center', fontsize=9)

    ax.set_title(region_name + " 연령대별 인구 분포 (2026년 1월)",
                fontsize=14, fontweight='bold', pad=15)
    ax.set_xlabel('인구수 (명)')
    ax.set_ylabel('연령대')
    ax.grid(axis='x', alpha=0.3)
    plt.tight_layout()
    plt.show()

📚 이번 프로젝트 핵심 정리

encoding='cp949'

str.contains()

3,915개 행에서 이름 포함 검색

iterrows()

f'2026년01월_계_{i}세'

f-string으로 컬럼명 자동 생성

split('(')[0].strip()

괄호 앞까지만 잘라 지역명 정리

sum(pop[s:e])
