📌 Series란?

Series = 엑셀의 '한 열' 입니다. 값들의 목록과 각 값에 붙는 번호(인덱스)로 구성됩니다. 번호 대신 이름을 인덱스로 사용할 수도 있습니다.

02_series.py 
# 학생들의 수학 점수로 Series를 만들어봅니다
math_scores = pd.Series([85, 92, 78, 95, 88])

print("=== 수학 점수 Series ===")
print(math_scores)

# 인덱스를 이름으로 지정할 수도 있습니다
math_scores_named = pd.Series(
    [85, 92, 78, 95, 88],
    index=['김민준', '이서연', '박지호', '최예린', '정도윤']
)

# 특정 학생 점수 조회
print(f"이서연 점수: {math_scores_named['이서연']}점")

# 기본 통계 계산
print(f"평균: {math_scores_named.mean():.1f}점")
print(f"최고: {math_scores_named.max()}점")

데이터 선택하기 (loc & iloc)
원하는 행/열만 골라내는 방법
🔍

loc = 이름으로 찾기 (「김민준 어디있어?」)
iloc = 번호로 찾기 (「3번 자리 학생!」) — i는 integer(정수)의 약자

05_select.py 
## 열(컬럼) 선택 ##
# 열 하나 선택 → Series 반환
df['수학']

# 여러 열 선택 → DataFrame 반환 (대괄호 두 겹!)
df[['이름', '수학', '영어']]

## 행 선택 ##
# iloc - 위치(번호)로 선택 (0부터 시작!)
df.iloc[0]          # 첫 번째 행
df.iloc[1:4]       # 1~3번 행 (4번 미포함)
df.iloc[0:3, 0:3]  # 0~2행, 0~2열

# loc - 인덱스 이름으로 선택
df.loc[0]              # 인덱스가 0인 행
df.loc[0:2, ['이름', '수학']]  # loc은 끝 번호 포함!
ℹ️

iloc vs loc의 슬라이싱 차이:
iloc[1:4] → 1, 2, 3번 (4번 미포함, 파이썬 일반 규칙)
loc[1:3] → 1, 2, 3번 (3번 포함! loc은 양쪽 다 포함)

결측값 확인 및 처리
NaN(Not a Number) - 값이 없는 빈 칸 다루기
01_missing_values.py 
import pandas as pd
import numpy as np

# 결측값이 포함된 데이터 만들기 (np.nan = 값 없음)
data = {
    '이름':  ['김민준', '이서연', '박지호', '최예린', '정도윤'],
    '국어':  [85, 92, 78, np.nan, 88],   # 최예린 국어 결측
    '수학':  [92, 78, np.nan, 88, 72],   # 박지호 수학 결측
    '영어':  [88, np.nan, 82, 91, 85]    # 이서연 영어 결측
}
df = pd.DataFrame(data)

## 결측값 확인 ##
# isnull() → 각 값이 결측인지 True/False
df.isnull()

# 열별 결측값 개수
print(df.isnull().sum())

# 결측값 비율 계산
missing_ratio = (df.isnull().sum() / len(df) * 100).round(1)
print(missing_ratio)

## 결측값 처리 ##

# 방법 1: 결측값 있는 행 삭제
df_dropped = df.dropna()

# 방법 2: 평균으로 채우기
df_filled = df.copy()  # 원본 보존을 위해 복사
for col in ['국어', '수학', '영어']:
    col_mean = df_filled[col].mean()
    df_filled[col] = df_filled[col].fillna(col_mean).round(1)

# 방법 3: 특정 값으로 채우기
df.fillna(0)        # 0으로 채우기
df.fillna('없음')   # 문자로 채우기

# 특정 값 포함 여부: isin()

# 반도체 공정 센서 데이터 생성 (100개 샘플)
np.random.seed(42)
n = 100

semi_data = {
    '공정ID':     [f'LOT{str(i).zfill(3)}' for i in range(1, n+1)],
    '온도(°C)':   np.round(np.random.normal(300, 15, n), 1),
    '습도(%)':    np.round(np.random.normal(45,  5,  n), 1),
    '압력(Pa)':   np.round(np.random.normal(101.3, 2, n), 2),
    '가스유량':   np.round(np.random.normal(100, 8, n), 1),
    '두께(nm)':   np.round(np.random.normal(50, 3, n), 2),
    '불량여부':   np.random.choice([0, 1], n, p=[0.85, 0.15])
}
df_semi = pd.DataFrame(semi_data)

# 기본 정보 확인
df_semi.info()
df_semi.describe()

# 정상 vs 불량 비교
result = df_semi.groupby('불량여부')[['온도(°C)', '두께(nm)']].mean()

# 불량률 계산
defect_rate = df_semi['불량여부'].mean() * 100
print(f"불량률: {defect_rate:.1f}%")

months = ['1월', '2월', '3월', '4월', '5월', '6월',

히스토그램 (Histogram)
데이터의 분포 - "몇 점대에 학생이 몇 명?"
03_histogram.py 
fig, axes = plt.subplots(1, 3, figsize=(15, 5))
colors = ['#FF6B6B', '#4ECDC4', '#45B7D1']

for ax, col, color in zip(axes, ['국어', '수학', '영어'], colors):
    ax.hist(df[col], bins=15, color=color,
            edgecolor='white', alpha=0.85)

    # 평균선 추가
    ax.axvline(df[col].mean(), color='black',
               linestyle='--', linewidth=2,
               label=f"평균: {df[col].mean():.1f}")

    ax.set_title(f'{col} 점수 분포', fontsize=14, fontweight='bold')
    ax.set_xlabel('점수'); ax.set_ylabel('학생 수')
    ax.legend(); ax.grid(alpha=0.3)

plt.tight_layout(); plt.show()

산점도 (Scatter Plot)
두 변수 사이의 관계(상관관계)를 시각화
04_scatter.py 
fig, ax = plt.subplots(figsize=(10, 7))

# 정상/불량을 다른 색과 마커로 표시
normal = df_semi[df_semi['불량여부'] == 0]
defect = df_semi[df_semi['불량여부'] == 1]

ax.scatter(normal['온도(°C)'], normal['두께(nm)'],
           c='#27AE60', alpha=0.7, s=80,
           marker='o', label='✅ 정상 (0)')
ax.scatter(defect['온도(°C)'], defect['두께(nm)'],
           c='#E74C3C', alpha=0.9, s=100,
           marker='X', label='❌ 불량 (1)')

# 상관계수 계산
corr = df_semi['온도(°C)'].corr(df_semi['두께(nm)'])
ax.set_title(f'온도 vs 두께 (상관계수: {corr:.3f})', fontsize=15)
ax.set_xlabel('온도 (°C)'); ax.set_ylabel('두께 (nm)')
ax.legend(); ax.grid(alpha=0.3)
plt.tight_layout(); plt.show()

종합 대시보드 만들기
subplot으로 여러 그래프를 한 화면에 배치
05_dashboard.py 
fig = plt.figure(figsize=(16, 12))
fig.suptitle('📊 학생 성적 분석 대시보드', fontsize=20, fontweight='bold')

# 2행 2열 배치 (2,2,1) = 2행 2열 중 첫 번째 위치
ax1 = fig.add_subplot(2, 2, 1)  # 왼쪽 위
ax2 = fig.add_subplot(2, 2, 2)  # 오른쪽 위
ax3 = fig.add_subplot(2, 2, 3)  # 왼쪽 아래
ax4 = fig.add_subplot(2, 2, 4)  # 오른쪽 아래

# 각 ax에 원하는 그래프를 그립니다
ax1.bar(df['이름'], df['평균'], color='#3498DB')
ax1.set_title('학생별 평균')

# ... (ax2, ax3, ax4도 같은 방식으로)

# 파일로 저장 (dpi가 높을수록 고화질)
plt.tight_layout()
plt.savefig('dashboard.png', dpi=150, bbox_inches='tight')
plt.show()
💾

savefig()로 파일 저장: 그래프를 PNG, JPG, PDF 등으로 저장할 수 있습니다. dpi=150은 해상도(높을수록 선명), bbox_inches='tight'는 여백 자동 조정입니다.

📚 필수 코드 패턴 - 그래프 꾸미기

template.py 
## 그래프 꾸미기 필수 요소 ##
plt.title('제목', fontsize=16, fontweight='bold')  # 제목
plt.xlabel('x축 레이블', fontsize=12)              # x축 이름
plt.ylabel('y축 레이블', fontsize=12)              # y축 이름
plt.legend(fontsize=11)                            # 범례 표시
plt.grid(alpha=0.3)                                # 격자선
plt.xlim(0, 100)                                   # x축 범위
plt.ylim(0, 100)                                   # y축 범위
plt.xticks(rotation=45)                            # x축 레이블 45도 회전
plt.tight_layout()                                 # 레이아웃 자동 조정
plt.savefig('graph.png', dpi=150)                 # 파일로 저장
plt.show()                                         # 화면 출력
